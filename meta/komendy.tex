\newcommand\img[4]{
	\begin{figure}[H]
		\begin{center}
			\includegraphics[width=#4\textwidth]{#1}
			\caption{#2}
			\label{fig:#3}
		\end{center}
	\end{figure}
} 

\newcommand\tabela[4]{
	\begin{table}[!h]
		\begin{center}
			\bgroup
			\def\arraystretch{1.5}
			\begin{tabular}{ #2 }
				#1
			\end{tabular}
			\egroup
		\end{center}
		\caption{#3}
		\label{tab:#4}
	\end{table}
}

\newcommand\codeinline[1]{\lstinline[basicstyle=\normalsize\ttfamily]{#1}}

\makeatletter
\newcommand{\sectionauthor}[1]{
{\vspace*{-5pt}\hspace*{4pt}\footnotesize\itshape\MakeUppercase{#1}
\par\nobreak\vspace*{10pt}}
\@afterheading}
\makeatother
